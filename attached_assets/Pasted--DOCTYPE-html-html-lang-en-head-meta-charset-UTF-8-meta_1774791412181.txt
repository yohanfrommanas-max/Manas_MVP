<script>
const ROUNDS = 5;
const MEMO_MS = 5000;

let targetH, targetS, targetL;
let guessH, guessS, guessL;
let round = 0;
let scores = [];
let memoTimer = null;

function hsl(h, s, l) { return `hsl(${h},${s}%,${l}%)`; }

function rand(min, max) { return Math.round(min + Math.random() * (max - min)); }

function showPhase(id) {
  document.querySelectorAll('.phase').forEach(p => p.classList.remove('active'));
  document.getElementById(id).classList.add('active');
}

function updateRoundLabels() {
  const txt = `Round ${round} of ${ROUNDS}`;
  const scoresTxt = scores.length ? `Best: ${Math.max(...scores)}%` : '';
  ['memo-round','match-round','result-round'].forEach(id => {
    const el = document.getElementById(id);
    if (el) el.textContent = txt;
  });
  ['memo-scores','match-scores','result-scores'].forEach(id => {
    const el = document.getElementById(id);
    if (el) el.textContent = scoresTxt;
  });
}

function startRound() {
  round++;
  targetH = rand(0, 359);
  targetS = rand(35, 90);
  targetL = rand(25, 65);
  guessH = rand(0, 359);
  guessS = 50;
  guessL = 50;
  updateRoundLabels();
  document.getElementById('target-swatch').style.background = hsl(targetH, targetS, targetL);
  showPhase('ph-memo');
  startMemoTimer();
}

function startMemoTimer() {
  const bar = document.getElementById('memo-bar');
  bar.style.transition = 'none';
  bar.style.width = '100%';
  requestAnimationFrame(() => requestAnimationFrame(() => {
    bar.style.transition = `width ${MEMO_MS}ms linear`;
    bar.style.width = '0%';
  }));
  clearTimeout(memoTimer);
  memoTimer = setTimeout(goToMatch, MEMO_MS);
}

function goToMatch() {
  showPhase('ph-match');
  renderWheels();
  updateGuessSwatch();
}

function resizeCanvas(c) {
  const w = c.parentElement.clientWidth;
  if (c.width !== w) c.width = w;
}

function drawHueWheel(c) {
  resizeCanvas(c);
  const ctx = c.getContext('2d');
  const w = c.width, h = c.height;
  for (let x = 0; x < w; x++) {
    ctx.fillStyle = hsl(Math.round((x / w) * 360), guessS, guessL);
    ctx.fillRect(x, 0, 1, h);
  }
}

function drawSatWheel(c) {
  resizeCanvas(c);
  const ctx = c.getContext('2d');
  const w = c.width, h = c.height;
  for (let x = 0; x < w; x++) {
    ctx.fillStyle = hsl(guessH, Math.round((x / w) * 100), guessL);
    ctx.fillRect(x, 0, 1, h);
  }
}

function drawLitWheel(c) {
  resizeCanvas(c);
  const ctx = c.getContext('2d');
  const w = c.width, h = c.height;
  for (let x = 0; x < w; x++) {
    ctx.fillStyle = hsl(guessH, guessS, Math.round((x / w) * 100));
    ctx.fillRect(x, 0, 1, h);
  }
}

function positionThumb(thumbId, fraction) {
  const thumb = document.getElementById(thumbId);
  const wrap = thumb.parentElement;
  const w = wrap.clientWidth;
  const px = Math.max(11, Math.min(w - 11, fraction * w));
  thumb.style.left = px + 'px';
  thumb.style.background = hsl(guessH, guessS, guessL);
}

function renderWheels() {
  drawHueWheel(document.getElementById('hue-canvas'));
  drawSatWheel(document.getElementById('sat-canvas'));
  drawLitWheel(document.getElementById('lit-canvas'));
  positionThumb('hue-thumb', guessH / 360);
  positionThumb('sat-thumb', guessS / 100);
  positionThumb('lit-thumb', guessL / 100);
  document.getElementById('hue-val').textContent = Math.round(guessH) + '°';
  document.getElementById('sat-val').textContent = Math.round(guessS) + '%';
  document.getElementById('lit-val').textContent = Math.round(guessL) + '%';
}

function updateGuessSwatch() {
  document.getElementById('guess-swatch').style.background = hsl(guessH, guessS, guessL);
}

function getFraction(e, wrap) {
  const rect = wrap.getBoundingClientRect();
  const x = (e.touches ? e.touches[0].clientX : e.clientX) - rect.left;
  return Math.max(0, Math.min(1, x / rect.width));
}

function setupWheel(canvasId, wrapId, onDrag) {
  const canvas = document.getElementById(canvasId);
  const wrap = document.getElementById(wrapId);
  let active = false;

  const down = e => { active = true; e.preventDefault(); onDrag(getFraction(e, wrap)); };
  const move = e => { if (!active) return; e.preventDefault(); onDrag(getFraction(e, wrap)); };
  const up = () => { active = false; };

  canvas.addEventListener('mousedown', down);
  canvas.addEventListener('touchstart', down, { passive: false });
  window.addEventListener('mousemove', move);
  window.addEventListener('touchmove', move, { passive: false });
  window.addEventListener('mouseup', up);
  window.addEventListener('touchend', up);
}

setupWheel('hue-canvas', 'hue-wrap', f => {
  guessH = Math.round(f * 360);
  renderWheels();
  updateGuessSwatch();
});

setupWheel('sat-canvas', 'sat-wrap', f => {
  guessS = Math.round(f * 100);
  renderWheels();
  updateGuessSwatch();
});

setupWheel('lit-canvas', 'lit-wrap', f => {
  guessL = Math.round(f * 100);
  renderWheels();
  updateGuessSwatch();
});

function hueDiff(a, b) {
  const d = Math.abs(a - b);
  return Math.min(d, 360 - d);
}

function calcScore() {
  const dH = hueDiff(targetH, guessH);
  const dS = Math.abs(targetS - guessS);
  const dL = Math.abs(targetL - guessL);
  const hScore = Math.max(0, 100 - (dH / 180) * 100);
  const sScore = Math.max(0, 100 - dS);
  const lScore = Math.max(0, 100 - dL);
  const total = Math.round(hScore * 0.5 + sScore * 0.25 + lScore * 0.25);
  return { total, dH: Math.round(dH), dS: Math.round(dS), dL: Math.round(dL) };
}

function badge(v, unit, isHue) {
  const threshold = isHue ? [10, 30] : [5, 20];
  let cls, label;
  if (v === 0) { cls = 'badge-perfect'; label = 'Perfect'; }
  else if (v <= threshold[0]) { cls = 'badge-close'; label = 'Close'; }
  else { cls = 'badge-off'; label = 'Off'; }
  return `${v}${unit} off <span class="breakdown-badge ${cls}">${label}</span>`;
}

document.getElementById('submit-btn').addEventListener('click', () => {
  clearTimeout(memoTimer);
  const { total, dH, dS, dL } = calcScore();
  scores.push(total);

  document.getElementById('res-target').style.background = hsl(targetH, targetS, targetL);
  document.getElementById('res-guess').style.background = hsl(guessH, guessS, guessL);
  document.getElementById('round-score-num').textContent = total + '%';
  document.getElementById('d-hue-text').innerHTML = badge(dH, '°', true);
  document.getElementById('d-sat-text').innerHTML = badge(dS, '%', false);
  document.getElementById('d-lit-text').innerHTML = badge(dL, '%', false);

  updateRoundLabels();
  showPhase('ph-result');
});

document.getElementById('next-btn').addEventListener('click', () => {
  if (round >= ROUNDS) {
    showFinal();
  } else {
    startRound();
  }
});

function gradeText(avg) {
  if (avg >= 95) return 'Exceptional colour memory';
  if (avg >= 85) return 'Strong colour perception';
  if (avg >= 72) return 'Good eye — keep training';
  if (avg >= 58) return 'Getting there';
  return 'Colour memory needs work';
}

function showFinal() {
  const avg = Math.round(scores.reduce((a, b) => a + b, 0) / scores.length);
  document.getElementById('final-avg').textContent = avg + '%';
  document.getElementById('final-grade').textContent = gradeText(avg);

  const hist = document.getElementById('hist-row');
  hist.innerHTML = '';
  scores.forEach((s, i) => {
    const pill = document.createElement('div');
    pill.className = 'hist-pill';
    pill.textContent = `R${i + 1}: ${s}%`;
    const hue = Math.round(s * 1.2);
    pill.style.background = `hsla(${hue}, 55%, 35%, 0.25)`;
    pill.style.borderColor = `hsla(${hue}, 55%, 55%, 0.3)`;
    pill.style.color = `hsl(${hue}, 60%, 70%)`;
    hist.appendChild(pill);
  });

  showPhase('ph-final');
}

document.getElementById('play-again-btn').addEventListener('click', () => {
  scores = [];
  round = 0;
  startRound();
});

document.getElementById('share-btn').addEventListener('click', () => {
  const avg = Math.round(scores.reduce((a, b) => a + b, 0) / scores.length);
  const breakdown = scores.map((s, i) => `R${i + 1}: ${s}%`).join(' · ');
  const text = `Colour Match — ${avg}% accuracy\n${breakdown}\nPlay at manas.app`;
  navigator.clipboard.writeText(text).then(() => {
    const btn = document.getElementById('share-btn');
    btn.textContent = 'Copied!';
    setTimeout(() => btn.textContent = 'Copy result', 2000);
  });
});

document.getElementById('start-btn').addEventListener('click', () => startRound());

const icon = document.getElementById('start-icon');
const colours = ['hsl(200,70%,50%)', 'hsl(340,65%,60%)', 'hsl(40,80%,55%)', 'hsl(160,60%,45%)'];
